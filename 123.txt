#Карточка товара
class CardGuest(QWidget):
    def __init__(self, id, photo, name, category, type_yslugi, opisanie, proizvoditel, postavchik, price, edinica, ostatok, sale):
        super().__init__()

        self.setMinimumHeight(170)

        if ostatok == 0:
            self.setStyleSheet("background-color: #ADD8E6; color: black")
        elif sale > 15:
            self.setStyleSheet("background-color: #2E8B57; color: black")

        layout = QHBoxLayout(self)

        self.label = QLabel()
        pixmap = QPixmap(photo)
        if pixmap.isNull():
            pixmap = QPixmap("images/picture.png")
        pixmap = pixmap.scaled(150, 150)
        self.label.setPixmap(pixmap)
        layout.addWidget(self.label)

        info = QVBoxLayout()
        info.addWidget(QLabel(name))
        info.addWidget(QLabel(f"Категория: {category}"))
        info.addWidget(QLabel(f"Тип: {type_yslugi}"))
        info.addWidget(QLabel(f"Описание: {opisanie}"))
        info.addWidget(QLabel(f"Производитель: {proizvoditel}"))
        info.addWidget(QLabel(f"Поставщик: {postavchik}"))
        info.addWidget(QLabel(f"Единица: {edinica}"))
        info.addWidget(QLabel(f"На складе: {ostatok}"))
        layout.addLayout(info)

        money = QVBoxLayout()
        self.priceLabel = QLabel()
        if sale > 0:
            final = round(float(price) * (1 - sale / 100), 2)
            self.priceLabel.setText(f'<s style="color:red">{price} руб.</s> '
                                    f'<span style="color:black">{final} руб.</span>')
        else:
            self.priceLabel.setText(f"Цена: {price} рублей")
        money.addWidget(self.priceLabel)
        money.addWidget(QLabel(f"Скидка: {sale} %"))
        if ostatok == 0:
            money.addWidget(QLabel("Нет на складе"))
        layout.addLayout(money)

    def auth(self):
        login = self.lineEdit.text()
        password = self.lineEdit_2.text()

        try:
            db = connectDB()
            cur = db.cursor()
            cur.execute("SELECT r.name, u.name FROM users u JOIN role r ON u.role_id = r.id "
                        "WHERE u.login = %s and u.password = %s", (login, password))
            user = cur.fetchone()
        except:
            QMessageBox.critical(self, "Ошибка", "Не удалось подключиться к базе данных")
            return

        if user is None:
            QMessageBox.warning(self, "Ошибка входа", "Пользователь не найден, войдите как гость")
            return

        role = user[0]
        name = user[1]

        if role == "client":
            self.window = clientWindow(name)
            self.window.show()
            self.close()

        elif role == "manager":
            self.window = managerWindow(name)
            self.window.show()
            self.close()

        elif role == "admin":
            self.window = adminWindow(name)
            self.window.show()
            self.close()

    def authGuest(self):
        self.window = GuestWindow()
        self.window.show()
        self.close()


PRODUCT_SQL = ("SELECT p.id, p.photo, p.name, c.name, p.type_yslugi, p.opisanie, "
               "m.name, s.name, p.price, u.name, p.ostatok, p.sale FROM product p "
               "LEFT JOIN category c ON p.category_id = c.id "
               "LEFT JOIN manufacturer m ON p.manufacturer_id = m.id "
               "LEFT JOIN supplier s ON p.supplier_id = s.id "
               "LEFT JOIN unit u ON p.unit_id = u.id")

    #Загрузка товаров с поиском, фильтром и сортировкой
    def createCard(self):
        search = self.searchEdit.text()
        filter = self.filterBox.currentText()
        sort = self.sortBox.currentText()

        db = connectDB()
        cur = db.cursor()

        sql = PRODUCT_SQL + " WHERE 1=1"
        params = []

        if search != "":
            sql += " AND (LOWER(p.name) LIKE LOWER(%s) OR LOWER(c.name) LIKE LOWER(%s) OR LOWER(p.type_yslugi) LIKE LOWER(%s) OR LOWER(p.opisanie) LIKE LOWER(%s) OR LOWER(m.name) LIKE LOWER(%s) OR LOWER(s.name) LIKE LOWER(%s) OR LOWER(u.name) LIKE LOWER(%s))"
            like = f"%{search}%"
            params += [like, like, like, like, like, like, like]

        if filter != "Все поставщики":
            sql += " AND s.name = %s"
            params.append(filter)

        if sort == "По возрастанию остатка":
            sql += " ORDER BY p.ostatok ASC"
        elif sort == "По убыванию остатка":
            sql += " ORDER BY p.ostatok DESC"

        cur.execute(sql, params)
        self.product = cur.fetchall()

        self.listWidget.clear()
        for row in self.product:
            card = CardGuest(*row)
            item = QListWidgetItem()
            item.setSizeHint(card.sizeHint())
            self.listWidget.addItem(item)
            self.listWidget.setItemWidget(item, card)

    #Загрузка заказов
    def createOrder(self):
        db = connectDB()
        cur = db.cursor()
        cur.execute("SELECT o.id, o.article, p.name, st.name, o.address, o.order_date, o.delivery_date "
                    "FROM orders o LEFT JOIN product p ON o.product_id = p.id "
                    "LEFT JOIN order_status st ON o.status_id = st.id")

            item = QListWidgetItem(text)

    def loadBoxes(self):
        db = connectDB()
        cur = db.cursor()
        for table, box in [("category", self.categoryBox), ("manufacturer", self.proizvoditelBox),
                           ("supplier", self.postavchikBox), ("unit", self.edinicaBox)]:
            cur.execute(f"SELECT id, name FROM {table}")
            for row in cur.fetchall():
                box.addItem(row[1], row[0])

    def load(self):
        db = connectDB()
        cur = db.cursor()
        cur.execute("SELECT photo, name, category_id, type_yslugi, opisanie, manufacturer_id, supplier_id, price, unit_id, ostatok, sale FROM product WHERE id = %s", (self.id,))
        row = cur.fetchone()

        self.photo = row[0]
        self.nameEdit.setText(row[1])
        self.categoryBox.setCurrentIndex(self.categoryBox.findData(row[2]))
        self.typeEdit.setText(row[3])
        self.opisanieEdit.setText(row[4])
        self.proizvoditelBox.setCurrentIndex(self.proizvoditelBox.findData(row[5]))
        self.postavchikBox.setCurrentIndex(self.postavchikBox.findData(row[6]))
        self.priceEdit.setValue(float(row[7]))
        self.edinicaBox.setCurrentIndex(self.edinicaBox.findData(row[8]))
        self.ostatokEdit.setValue(row[9])
        self.saleEdit.setValue(row[10])

        pixmap = QPixmap(self.photo)
        if pixmap.isNull():
            pixmap = QPixmap("images/picture.png")
        self.photoLabel.setPixmap(pixmap.scaled(150, 100))

    def pickPhoto(self):
        path, _ = QFileDialog.getOpenFileName(self, "Выберите фото", "", "Изображения (*.png *.jpg *.jpeg)")
        if path:
            self.newPhoto = path
            self.photoLabel.setPixmap(QPixmap(path).scaled(150, 100))

    def savePhoto(self):
        if self.newPhoto is None:
            return self.photo
        pixmap = QPixmap(self.newPhoto)
        pixmap = pixmap.scaled(300, 200)
        os.makedirs("images", exist_ok=True)
        name = "img_" + uuid.uuid4().hex[:8] + ".png"
        path = "images/" + name
        pixmap.save(path)
        if self.photo and self.photo.startswith("images/img_") and os.path.exists(self.photo):
            os.remove(self.photo)
        return path

    #Сохранение товара
    def save(self):
        if self.nameEdit.text() == "":
            QMessageBox.warning(self, "Предупреждение", "Введите наименование товара")
            return

        try:
            photo = self.savePhoto()
            data = (photo, self.nameEdit.text(), self.categoryBox.currentData(), self.typeEdit.text(),
                    self.opisanieEdit.text(), self.proizvoditelBox.currentData(), self.postavchikBox.currentData(),
                    self.priceEdit.value(), self.edinicaBox.currentData(), self.ostatokEdit.value(), self.saleEdit.value())

            db = connectDB()
            cur = db.cursor()
            if self.id is None:

        cur = db.cursor()

        layout.addWidget(self.productBox)

        layout.addWidget(QLabel("Артикул:"))
        self.articleEdit = QLineEdit()
        layout.addWidget(self.articleEdit)

        layout.addWidget(QLabel("Статус заказа:"))
        self.statusBox = QComboBox()
        cur.execute("SELECT id, name FROM order_status")
        for row in cur.fetchall():
            self.statusBox.addItem(row[1], row[0])
        layout.addWidget(self.statusBox)

        layout.addWidget(QLabel("Адрес пункта выдачи:"))
        self.addressEdit = QLineEdit()
        layout.addWidget(self.addressEdit)

        layout.addWidget(QLabel("Дата заказа:"))
        self.orderDate = QDateEdit(QDate.currentDate())
        self.orderDate.setCalendarPopup(True)
        layout.addWidget(self.orderDate)

        layout.addWidget(QLabel("Дата выдачи:"))
        self.deliveryDate = QDateEdit(QDate.currentDate())
        self.deliveryDate.setCalendarPopup(True)
        layout.addWidget(self.deliveryDate)

        buttons = QHBoxLayout()
        self.saveButton = QPushButton("Сохранить")
        self.saveButton.setStyleSheet(ACCENT_STYLE)
        self.saveButton.clicked.connect(self.save)
        buttons.addWidget(self.saveButton)
        if id is not None:
            self.deleteButton = QPushButton("Удалить")
            self.deleteButton.clicked.connect(self.delete)
            buttons.addWidget(self.deleteButton)
        layout.addLayout(buttons)

        if id is not None:
            self.load()

    def load(self):
        db = connectDB()
        cur = db.cursor()
        cur.execute("SELECT article, product_id, status_id, address, order_date, delivery_date FROM orders WHERE id = %s", (self.id,))
        row = cur.fetchone()

        self.articleEdit.setText(row[0])
        self.productBox.setCurrentIndex(self.productBox.findData(row[1]))
        self.statusBox.setCurrentIndex(self.statusBox.findData(row[2]))
        self.addressEdit.setText(row[3])
        if row[4]:
            self.orderDate.setDate(QDate(row[4].year, row[4].month, row[4].day))
        if row[5]:
            self.deliveryDate.setDate(QDate(row[5].year, row[5].month, row[5].day))

    #Сохранение заказа
    def save(self):
        if self.articleEdit.text() == "":
            QMessageBox.warning(self, "Предупреждение", "Введите артикул заказа")
            return

        try:
            data = (self.articleEdit.text(), self.productBox.currentData(), self.statusBox.currentData(),
                    self.addressEdit.text(), self.orderDate.date().toString("yyyy-MM-dd"),
                    self.deliveryDate.date().toString("yyyy-MM-dd"))

            db = connectDB()
            cur = db.cursor()
            if self.id is None:
                cur.execute("INSERT INTO orders(article, product_id, status_id, address, order_date, delivery_date) VALUES (%s,%s,%s,%s,%s,%s)", data)
            else:
                cur.execute("UPDATE orders SET article=%s, product_id=%s, status_id=%s, address=%s, order_date=%s, delivery_date=%s WHERE id=%s", data + (self.id,))
            db.commit()
        except:
            QMessageBox.critical(self, "Ошибка", "Не удалось сохранить заказ")
            return

        self.adminWindow.createOrder()
        self.close()

    def delete(self):
        box = QMessageBox(self)
        box.setWindowTitle("Подтверждение")
        box.setText("Удалить заказ?")
        yes = box.addButton("Да", QMessageBox.ButtonRole.YesRole)
        box.addButton("Нет", QMessageBox.ButtonRole.NoRole)
        box.exec()
        if box.clickedButton() == yes:
            db = connectDB()
            cur = db.cursor()
            cur.execute("DELETE FROM orders WHERE id = %s", (self.id,))
            db.commit()
            self.adminWindow.createOrder()
            self.close()
